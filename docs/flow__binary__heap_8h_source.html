174 // Fix down by repeatedly choosing the less child.
175 // Return the index of the hole at the bottom.
176 std::size_t fixDownWithHole(std::size_t index)
177 noexcept(std::is_nothrow_move_assignable_v<T>) {
178 for (;;) {
179 size_t child = getLeftChild(index);
180 if (child >= data_.size()) {
181 break;
182 }
183
184 // Select the less child.
185 std::size_t rightChild = child + 1;
186 if (rightChild < data_.size() && comparator_(data_[rightChild], data_[child])) {
187 child = rightChild;
188 }
189 data_[index] = std::move(data_[child]);
190 index = child;
191 }
192 return index;
193 }

195 // Fix up the value.
196 void fixUp(std::size_t index, std::size_t top, T&& value)
197 noexcept(std::is_nothrow_move_assignable_v<T>) {
198 while (index > top) {
199 std::size_t parent = getParent(index);
200 if (comparator_(value, data_[parent])) {
